app/(dashboard)/checkouts/create/
├── page.tsx                        # Server Component — teams/equipment 데이터 prefetch, ~30줄

│   └── StepIndicator (내부 컴포넌트)
└── steps/
    ├── PurposeStep.tsx             # Step 1 — 목적 카드 3개 + 대여 확장 필드, ~150줄
    ├── EquipmentSelectionStep.tsx  # Step 2 — 검색 + 테이블 + 선택 카운터, ~180줄
    └── DetailsStep.tsx             # Step 3 — 폼 필드 + 선택 장비 칩 + 제출, ~180줄

# 상태 흐름: CreateCheckoutContent가 단일 wizard 상태 보유
# Step간 데이터 전달: { purpose, selectedEquipmentIds, formValues }
# URL sync: currentStep을 ?step=N으로 URL에 반영 (뒤로가기 지원)
